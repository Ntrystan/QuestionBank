    -----
    lijst = [3, 8, 12, -4]
    =====
    teller = 0
    =====
    getal = 0 #paired
    =====
    for getal in lijst:
    =====
    for teller in lijst: #paired
    =====
        teller += 1
    =====
        getal += 1 #paired
    =====
    print(teller)
    =====
    print(getal) #paired
    =====
    teller += getal #distractor
