    str = "The 0wl h00ts l0udly"
    # SET POS TO A VALUE GREATER THAN OR EQUAL TO 0
    pos = 1
    # SET WHILE CONDITION
    while pos >= 0:
        # REPLACE VALUE
        pos = str.find("0")
        if pos == -1:
          break
        str = str[0:pos] + "o" + str[pos+1:len(str)]
    # PRINT RESULT
    print(str)
