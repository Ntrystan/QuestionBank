    class Account:
        '''Account class to model bank accounts'''

        def __init__(self, name):
            '''Create a new account for owner name and a zero balance'''
            self.owner = name
            self.balance = 0.00

        def getBalance(self):
            return self.balance

        def setBalance(self, amount):
            self.balance = amount

    p = Account('Jan')
    print(p.getBalance())
    p.setBalance(150)
    print(p.getBalance())
